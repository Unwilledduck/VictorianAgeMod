country_event = {
	id = france.3
	title = france.3.t
	desc = france.3.d
	picture = GFX_report_event_german_parade_paris
	
	fire_only_once = yes

		tag = FRA
		has_war_with = PRS

		}
	}

		days = 2
	}

	immediate = {
		set_global_flag = achievement_france_surrender
	}
	
	option = {
		name = france.2.a
		set_global_flag = fall_of_france
	}

}
